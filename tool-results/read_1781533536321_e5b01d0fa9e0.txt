   261→   261→// Resolve team reference to actual team name
   262→   262→// e.g., "1A" -> group A winner, "2B" -> group B runner-up
   263→   263→export const resolveTeamRef = (
   264→   264→  ref: string,
   265→   265→  standings: Record<string, GroupStanding[]>,
   266→   266→  thirdPlaceRanking: { group: string; team: string }[],
   267→   267→  results: Record<number, { homeGoals: number; awayGoals: number; homePenalties?: number; awayPenalties?: number }>,
   268→   268→): string | null => {
   269→   269→  // Group winner: "1A", "1B", etc.
   270→   270→  const winnerMatch = ref.match(/^1([A-L])$/);
   271→   271→  if (winnerMatch) {
   272→   272→    const group = winnerMatch[1];
   273→   273→    const groupStandings = standings[group];
   274→   274→    if (groupStandings && groupStandings.length > 0 && groupStandings[0].played > 0) {
   275→   275→      return groupStandings[0].team;
   276→   276→    }
   277→   277→    return null;
   278→   278→  }
   279→   279→
   280→   280→  // Group runner-up: "2A", "2B", etc.
   281→   281→  const runnerUpMatch = ref.match(/^2([A-L])$/);
   282→   282→  if (runnerUpMatch) {
   283→   283→    const group = runnerUpMatch[1];
   284→   284→    const groupStandings = standings[group];
   285→   285→    if (groupStandings && groupStandings.length > 1 && groupStandings[1].played > 0) {
   286→   286→      return groupStandings[1].team;
   287→   287→    }
   288→   288→    return null;
   289→   289→  }
   290→   290→
   291→   291→  // Third place reference: "3ABCDf", "3CDFGH", etc.
   292→   292→  const thirdPlaceMatch = ref.match(/^3(.+)$/);
   293→   293→  if (thirdPlaceMatch) {
   294→   294→    const eligibleGroupsKey = ref;
   295→   295→    const eligibleGroups = THIRD_PLACE_ELIGIBLE_GROUPS[eligibleGroupsKey];
   296→   296→    if (!eligibleGroups) return null;
   297→   297→
   298→   298→    // Find the best 3rd place team from eligible groups (only if they've played matches)
   299→   299→    const eligibleThirdPlace = thirdPlaceRanking.filter(tp =>
   300→   300→      eligibleGroups.includes(tp.group) && tp.points > 0
   301→   301→    );
   302→   302→    if (eligibleThirdPlace.length > 0) {
   303→   303→      return eligibleThirdPlace[0].team;
   304→   304→    }
   305→   305→    return null;
   306→   306→  }
   307→   307→
   308→   308→  // Winner of a match: "W73", "W74", etc.
   309→   309→  const winnerOfMatch = ref.match(/^W(\d+)$/);
   310→   310→  if (winnerOfMatch) {
   311→   311→    const matchId = parseInt(winnerOfMatch[1]);
   312→   312→    const result = results[matchId];
   313→   313→    if (!result) return null;
   314→   314→    const match = MATCHES.find(m => m.id === matchId);
   315→   315→    if (!match) return null;
   316→   316→
   317→   317→    if (result.homeGoals > result.awayGoals) return match.team1;
   318→   318→    if (result.awayGoals > result.homeGoals) return match.team2;
   319→   319→    // Check penalties
   320→   320→    if (result.homePenalties !== undefined && result.awayPenalties !== undefined) {
   321→   321→      if (result.homePenalties > result.awayPenalties) return match.team1;
   322→   322→      if (result.awayPenalties > result.homePenalties) return match.team2;
   323→   323→    }
   324→   324→    return null;
   325→   325→  }
   326→   326→
   327→   327→  // Loser of a match: "L101", "L102", etc.
   328→   328→  const loserOfMatch = ref.match(/^L(\d+)$/);
   329→   329→  if (loserOfMatch) {
   330→   330→    const matchId = parseInt(loserOfMatch[1]);
   331→   331→    const result = results[matchId];
   332→   332→    if (!result) return null;
   333→   333→    const match = MATCHES.find(m => m.id === matchId);
   334→   334→    if (!match) return null;
   335→   335→
   336→   336→    if (result.homeGoals < result.awayGoals) return match.team1;
   337→   337→    if (result.awayGoals < result.homeGoals) return match.team2;
   338→   338→    // Check penalties
   339→   339→    if (result.homePenalties !== undefined && result.awayPenalties !== undefined) {
   340→   340→      if (result.homePenalties < result.awayPenalties) return match.team1;
   341→   341→      if (result.awayPenalties < result.homePenalties) return match.team2;
   342→   342→    }
   343→   343→    return null;
   344→   344→  }
   345→   345→
   346→   346→  return null;
   347→   347→};

   362→   362→    if (!eligibleGroups) return `أفضل ثالث`;

   364→   364→    const slots = ['3ABCDf','3CDFGH','3CEfHI','3EHIJK','3BEFIJ','3AEHIJ','3EFGIJ','3DEIJL'];
   365→   365→    const slotIndex = slots.indexOf(ref);
   366→   366→    const rankNum = slotIndex >= 0 ? slotIndex + 1 : '?';
   367→   367→    return `أفضل ثالث #${rankNum}`;
   368→   368→  }
   369→   369→  const winnerOfMatch = ref.match(/^W(\d+)$/);
   370→   370→  if (winnerOfMatch) {
   371→   371→    return `الفائز من مباراة ${winnerOfMatch[1]}`;
   372→   372→  }
   373→   373→  const loserOfMatch = ref.match(/^L(\d+)$/);
   374→   374→  if (loserOfMatch) {
   375→   375→    return `الخاسر من مباراة ${loserOfMatch[1]}`;
   376→   376→  }
